us_macro_quarterly = rename(us_macro_quarterly, TB10YS = GS10)
us_macro_quarterly = mutate(us_macro_quarterly,
 GDPGR = 400 * log(GDPC96/lag(GDPC96)),
 TSpread = TB10YS - TB3MS,
 date = date_decimal(Date,tz="UTC"),
 year = year(date))
